# BATCH_DEDUCTION_MERCHANT_MONTHLY_AGGREGATION

## 목적
[배치 개발 요청서] 배치명: 소득공제 월별 통합 집계 배치 기준: - 기준년월(BASE_YM) 대상 테이블: - TB_CARD_SALES_LEDGER - TB_BOOK_PERF_MERCHANT - TB_TRAD_MARKET_MERCHANT - TB_GENERAL_DEDUCT_MERCHANT 처리 내용: - 매출원장 기준으로 가맹점별 소득공제 대상 거래를 구분한다 - 도서공연 / 전통시장 / 일반 가맹점을 각각 분류한다 - 유효기간(APPLY_START_DT ~ APPLY_END_DT) 기준으로 가맹점 매칭한다 - 취소 거래 제외 - 고객별, 가맹점 유형별 월별 금액 집계 출력: - 고객ID, 기준년월, 가맹점유형, 총금액, 건수 조건: - CANCEL_YN = 'N' - USE_YN = 'Y'

## 생성 SQL
```sql
INSERT INTO TB_DEDUCTION_MONTHLY_SUMMARY (
    CUSTOMER_ID,
    BASE_YM,
    MERCHANT_TYPE,
    TOTAL_AMT,
    TXN_COUNT,
    REG_DTM
)
SELECT
    S.CUSTOMER_ID,
    S.BASE_YM,
    S.MERCHANT_TYPE,
    S.TOTAL_AMT,
    S.TXN_COUNT,
    NOW() AS REG_DTM
FROM (
SELECT
    A.CUSTOMER_ID AS CUSTOMER_ID,
    A.BASE_YM AS BASE_YM,
    CASE
    WHEN B.MERCHANT_ID IS NOT NULL THEN 'BOOK_PERF'
    WHEN T.MERCHANT_ID IS NOT NULL THEN 'TRAD_MARKET'
    WHEN G.MERCHANT_ID IS NOT NULL THEN 'GENERAL'
    ELSE 'UNKNOWN'
END AS MERCHANT_TYPE,
    SUM(A.SALES_AMT) AS TOTAL_AMT,
    COUNT(*) AS TXN_COUNT
FROM TB_CARD_SALES_LEDGER A
LEFT JOIN TB_BOOK_PERF_MERCHANT B
  ON A.MERCHANT_ID = B.MERCHANT_ID AND A.SALES_DT BETWEEN B.APPLY_START_DT AND IFNULL(B.APPLY_END_DT, '99991231')
 AND B.USE_YN = 'Y'
LEFT JOIN TB_TRAD_MARKET_MERCHANT T
  ON A.MERCHANT_ID = T.MERCHANT_ID AND A.SALES_DT BETWEEN T.APPLY_START_DT AND IFNULL(T.APPLY_END_DT, '99991231')
 AND T.USE_YN = 'Y'
LEFT JOIN TB_GENERAL_DEDUCT_MERCHANT G
  ON A.MERCHANT_ID = G.MERCHANT_ID AND A.SALES_DT BETWEEN G.APPLY_START_DT AND IFNULL(G.APPLY_END_DT, '99991231')
 AND G.USE_YN = 'Y'
WHERE A.BASE_YM = :base_ym
  AND A.CANCEL_YN = 'N'
  AND (B.MERCHANT_ID IS NOT NULL OR T.MERCHANT_ID IS NOT NULL OR G.MERCHANT_ID IS NOT NULL)
GROUP BY
    A.CUSTOMER_ID,
    A.BASE_YM,
    CASE
    WHEN B.MERCHANT_ID IS NOT NULL THEN 'BOOK_PERF'
    WHEN T.MERCHANT_ID IS NOT NULL THEN 'TRAD_MARKET'
    WHEN G.MERCHANT_ID IS NOT NULL THEN 'GENERAL'
    ELSE 'UNKNOWN'
END
) S
```
